cards[index].front : cards[index].back; } loadDeckList(); ``` --- ## flashcards.json (example deck) ```json { "title": "Sample Flashcards", "description": "Basic example deck for testing the system.", "cards": [ { "front": "What is the capital of France?", "back": "Paris" }, { "front": "2 + 2 = ?", "back": "4" }, { "front": "Who wrote '1984'?", "back": "George Orwell" } ] } ``` --- ## categories.json (new) ```json { "decks": [ { "name": "Sample Deck", "file": "flashcards.json" }, { "name": "Biology Basics", "file": "biology.json" }, { "name": "Spanish Verbs", "file": "spanish.json" } ] } ``` --- ## manifest.webmanifest (new) ```json { "name": "Flashcards App", "short_name": "Flashcards", "start_url": "index.html", "display": "standalone", "background_color": "#ffffff", "theme_color": "#000000", "icons": [] } ``` --- ## offline.html (new standalone offline version) ```html Offline Flashcards
